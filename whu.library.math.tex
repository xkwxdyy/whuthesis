  { \WHULoadLibrary { math-bachelor } }
  {

      { \WHULoadLibrary { math-master } }
      {

          { \WHULoadLibrary { math-doctor } }
          {
            % 如果没有加载 bachelor, master, doctor 库中的任意一个，则默认加载 math-bachelor
            \WHULoadLibrary { math-bachelor } 
          }
      }
  }
