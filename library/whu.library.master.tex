  }

  {
    degree-type .choice:,

    degree-type / professional .code:n = { \str_gset:Nn \g__whu_master_degree_type_str { professional } },
    degree-type .initial:n = academic,
    degree-type / unknown .code:n = { \msg_error:nnn { whu } { invalid-degree-type } {#1} },
  }
